let addedData = JSON.parse(localStorage.getItem("addedData_kita") || "[]");
let doneList = JSON.parse(localStorage.getItem("doneList_kita") || "[]");

function allData(){
  return DATA.concat(addedData);
}

function normalizeText(text){
  return String(text || "")
    .toLowerCase()
    .replace(/[０-９]/g, s => String.fromCharCode(s.charCodeAt(0)-65248))
    .replace(/　/g," ")
    .replace(/丁目/g,"")
    .replace(/番/g,"")
    .replace(/号/g,"");
}

function getFiltered(){
  const key = normalizeText(document.getElementById("keyword").value);
  const area = document.getElementById("area").value;

  return allData().filter(item=>{
    const target = normalizeText(
      item.name + " " + item.address + " " + item.tel + " " + item.area
    );
    const okKey = !key || target.includes(key);
    const okArea = !area || item.area === area;
    return okKey && okArea;
  });
}

function render(){
  const list = document.getElementById("list");
  const items = getFiltered();
  document.getElementById("count").textContent = items.length;
  list.innerHTML = "";

  if(items.length === 0){
    list.innerHTML = "<div class='card'>該当する営業先がありません。検索語を短くしてください。例：北区、天神橋、中崎、南森町、本庄、大淀</div>";
    return;
  }

  items.forEach((item, index)=>{
    const id = item.name + item.address;
    const done = doneList.includes(id);
    const telLink = item.tel ? `<a href="tel:${item.tel}"><button class="tel btn-small">電話</button></a>` : "";
    const mapUrl = "https://www.google.com/maps/search/?api=1&query=" + encodeURIComponent(item.address);
    const routeUrl = "https://www.google.com/maps/dir/?api=1&destination=" + encodeURIComponent(item.address) + "&travelmode=walking";

    const card = document.createElement("div");
    card.className = "card" + (done ? " done-card" : "");
    card.innerHTML = `
      <div class="name">${index+1}. ${item.name}</div>
      <div>
        <span class="badge">${item.area || "北区"}</span>
        <span class="badge">居宅介護支援</span>
      </div>
      <div class="meta">
        住所：${item.address}<br>
        電話：${item.tel || "未登録"}<br>
        FAX：${item.fax || "未登録"}
      </div>
      <div class="actions">
        ${telLink}
        <a href="${mapUrl}" target="_blank"><button class="map btn-small">地図</button></a>
        <a href="${routeUrl}" target="_blank"><button class="route btn-small">ここへ行く</button></a>
        <button class="done btn-small" onclick="toggleDone('${escapeForJs(id)}')">${done ? "営業済み解除" : "営業済み"}</button>
      </div>
    `;
    list.appendChild(card);
  });
}

function escapeForJs(str){
  return str.replace(/\\/g,"\\\\").replace(/'/g,"\\'");
}

function toggleDone(id){
  if(doneList.includes(id)){
    doneList = doneList.filter(x=>x!==id);
  }else{
    doneList.push(id);
  }
  localStorage.setItem("doneList_kita", JSON.stringify(doneList));
  render();
}

function resetSearch(){
  document.getElementById("keyword").value = "";
  document.getElementById("area").value = "";
  render();
}

function openRoute(){
  const items = getFiltered().slice(0,9);
  if(items.length === 0){
    alert("ルートにする営業先がありません");
    return;
  }

  const origin = "大阪市北区";
  const destination = items[items.length-1].address;
  const waypoints = items.slice(0, -1).map(x=>x.address).join("|");

  let url = "https://www.google.com/maps/dir/?api=1";
  url += "&origin=" + encodeURIComponent(origin);
  url += "&destination=" + encodeURIComponent(destination);
  if(waypoints){
    url += "&waypoints=" + encodeURIComponent(waypoints);
  }
  url += "&travelmode=walking";
  window.open(url, "_blank");
}

function addCSV(){
  const text = document.getElementById("csvInput").value.trim();
  if(!text){
    alert("CSVを入力してください");
    return;
  }

  const lines = text.split(/\n/);
  let added = 0;

  lines.forEach(line=>{
    const cols = line.split(",");
    if(cols.length >= 3){
      const item = {
        name: cols[0].trim(),
        address: cols[1].trim(),
        tel: cols[2].trim(),
        fax: "",
        area: cols[3] ? cols[3].trim() : "北区"
      };
      if(item.name && item.address){
        addedData.push(item);
        added++;
      }
    }
  });

  localStorage.setItem("addedData_kita", JSON.stringify(addedData));
  document.getElementById("csvInput").value = "";
  alert(added + "件追加しました");
  render();
}

function copyCSV(){
  const header = "事業所名,住所,電話番号,FAX,エリア\n";
  const rows = allData().map(x =>
    `"${x.name}","${x.address}","${x.tel || ""}","${x.fax || ""}","${x.area || ""}"`
  ).join("\n");

  navigator.clipboard.writeText(header + rows).then(()=>{
    alert("CSVをコピーしました。スプレッドシートやExcelに貼れます。");
  });
}

render();
